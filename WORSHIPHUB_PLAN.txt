--- 3A: PARSER AST (TypeScript) ---

// A single parsed line of a song
type SongSegment = {
  chord?: string;        // e.g. "Am7"
  text: string;          // lyric fragment, e.g. "бе"
};

type SongLine = {
  type: 'lyric' | 'cue' | 'empty';
  cue?: string;                    // if type='cue', e.g. "ИНТРО: ГИТАРА"
  segments?: SongSegment[];        // if type='lyric', chord+text pairs
};

type ParsedSong = {
  lines: SongLine[];
};

--- 3B: SUPABASE SCHEMA (Phase 1: No Auth) ---

-- Songs table (no user_id yet, added in auth phase)
CREATE TABLE songs (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  title TEXT NOT NULL,
  original_key TEXT,             -- e.g. "G", "Am"
  bpm INTEGER,
  content TEXT NOT NULL,         -- raw ChordPro Extended text
  tags TEXT[] DEFAULT '{}',      -- e.g. {"великий пост", "хвала"}
  created_at TIMESTAMPTZ DEFAULT now(),
  updated_at TIMESTAMPTZ DEFAULT now()
);

-- Setlists table
CREATE TABLE setlists (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  title TEXT NOT NULL,
  service_date DATE,
  notes TEXT,
  created_at TIMESTAMPTZ DEFAULT now(),
  updated_at TIMESTAMPTZ DEFAULT now()
);

-- Junction: songs in a setlist with per-setlist transposition
CREATE TABLE setlist_songs (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  setlist_id UUID REFERENCES setlists(id) ON DELETE CASCADE,
  song_id UUID REFERENCES songs(id) ON DELETE CASCADE,
  sort_order INTEGER NOT NULL,
  transpose_steps INTEGER DEFAULT 0,  -- semitones from original key
  capo_fret INTEGER DEFAULT 0
);


============================
  SECTION 4: PARSER SPEC
============================

File: src/features/songs/lib/parser.ts

INPUT: Raw ChordPro Extended string (multiline)
OUTPUT: ParsedSong (array of SongLine)

RULES:
1. Line starting with [! ...] only (no other text) → type: 'cue', block-level
2. Line containing [! ...] mixed with text → inline cue attached to segments
3. [X] where X is a valid chord → chord on next text segment
4. Bare text → lyric segment with no chord
5. Empty line → type: 'empty' (visual spacer)
6. Multiple chords before text → each creates a segment

PARSING EXAMPLE:

  Input:
    [! ИНТРО: ГИТАРА]
    [G] Слава Те[Em]бе, Боже
    [! ВСТУПАЮТ БАРАБАНЫ]
    [C] Ибо Ты ве[D]лик.

  Output:
    lines: [
      { type: 'cue', cue: 'ИНТРО: ГИТАРА' },
      { type: 'lyric', segments: [
          { chord: 'G', text: ' Слава Те' },
          { chord: 'Em', text: 'бе, Боже' }
      ]},
      { type: 'cue', cue: 'ВСТУПАЮТ БАРАБАНЫ' },
      { type: 'lyric', segments: [
          { chord: 'C', text: ' Ибо Ты ве' },
          { chord: 'D', text: 'лик.' }
      ]}
    ]

EDGE CASES TO HANDLE:
  - Slash chords: [G/B] → treat as single chord "G/B"
  - Chords with no text after: [G]  [Am] on a line alone → segments with empty text
  - Cyrillic and Latin mixed content
  - Lines with only lyrics (no chords) → segments with no chord field


============================
  SECTION 5: TRANSPOSER SPEC
============================

File: src/features/songs/lib/transposer.ts

CHROMATIC SCALE (sharps): C C# D D# E F F# G G# A A# B
CHROMATIC SCALE (flats):  C Db D Eb E F Gb G Ab A Bb B

FUNCTION: transposeChord(chord: string, steps: number): string

LOGIC:
1. Parse chord into: root + quality
   - Root: C, C#, Db, D, D#, Eb, E, F, F#, Gb, G, G#, Ab, A, A#, Bb, B
   - Quality: m, 7, m7, maj7, dim, aug, sus2, sus4, add9, etc. (everything after root)
   - Slash bass: if "/" present, split and transpose both sides
2. Find root index in chromatic scale
3. Add steps (mod 12)
4. Pick sharp or flat spelling based on TARGET KEY preference:
   - Sharp keys: G, D, A, E, B, F#, C → use sharps
   - Flat keys:  F, Bb, Eb, Ab, Db, Gb → use flats
5. Reattach quality

FUNCTION: transposeSong(content: string, steps: number): string
  - Regex find all [ChordHere] patterns (excluding [! ...] cues)
  - Apply transposeChord to each
  - Return modified string

ENHARMONIC MAP for display:
  { "C#": "Db", "D#": "Eb", "F#": "Gb", "G#": "Ab", "A#": "Bb" }

CAPO MODE:
  - If capo = 3 and song key = Bb:
    Display: "Каподастр: 3 | Играть как G"
  - Calculate: actual_key - capo_fret = displayed shapes

  SECTION 6: UI RENDERING

--- SongViewer.tsx: The Core Rendering Engine ---

For each SongLine:

IF type = 'cue':
  Render: <div class="text-sm text-blue-400 italic pl-2">🎵 {cue}</div>

IF type = 'empty':
  Render: <div class="h-4"></div>  (spacer)

IF type = 'lyric':
  Render a flex row where each segment is a column:
  ┌─────────────────────────────────────────┐
  │  [chord]     [chord]                    │  ← Row 1: Chords (green, bold)
  │  lyric text  lyric text                 │  ← Row 2: Lyrics (white, large)
  └─────────────────────────────────────────┘

  Each segment:
    <div class="inline-flex flex-col">
      <span class="text-green-400 font-bold text-sm">{chord}</span>
      <span class="text-white text-xl">{text}</span>
    </div>

--- Role Toggles ---
  - "Музыкант"  → show all 3 layers
  - "Певец"     → hide cues, show chords + lyrics
  - "Община"    → hide chords + cues, show only lyrics (large font)

--- Font Size ---
  Store in settingsStore. Apply via CSS variable or inline style.
  Range: 14px–36px, default 20px.

--- Auto-Scroll ---
  Configurable speed: 0.5x, 1x, 1.5x, 2x
  Toggle play/pause.
  Implementation: requestAnimationFrame scrolling the container.
